# Icon

Our component wrapper around SVG icons.

## Examples

export const DefaultTemplate = args => {
  return <Icon {...args} />;
};

export const ListTemplate = () => {
  return (
    <Box>
      {iconNames.map(icon => (
        <Box key={icon} display="inline-block" w="100px" m="20px" ta="center">
          <p>{icon}</p>
          <Icon name={icon} />
        </Box>
      ))}
    </Box>
  );
};

export const Default = DefaultTemplate.bind({});
