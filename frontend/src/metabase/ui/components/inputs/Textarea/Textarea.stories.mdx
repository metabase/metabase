# Textarea

Our themed wrapper around [Mantine Textarea](https://v6.mantine.dev/core/textarea/).

## Docs

- [Mantine Textarea Docs](https://v6.mantine.dev/core/textarea/)

## Examples

export const DefaultTemplate = args => <Textarea {...args} />;

export const VariantTemplate = args => (
  <Stack>
    <Textarea {...args} variant="default" />
    <Textarea {...args} variant="unstyled" />
  </Stack>
);
