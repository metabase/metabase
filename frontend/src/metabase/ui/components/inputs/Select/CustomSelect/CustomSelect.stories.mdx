import { Canvas, Story, Meta } from "@storybook/addon-docs";
import { SelectInput, Stack } from "metabase/ui";

export const args = {
  size: "md",
  placeholder: "No semantic type",
  clearable: false,
  disabled: false,
  readOnly: false,
};

export const sampleArgs = {
  value: "Entity key",
};

export const argTypes = {
  size: {
    options: ["xs", "md"],
    control: { type: "inline-radio" },
  },
  placeholder: {
    control: { type: "text" },
  },
  clearable: {
    control: { type: "boolean" },
  },
  disabled: {
    control: { type: "boolean" },
  },
  readOnly: {
    control: { type: "boolean" },
  },
};

<Meta
  title="Inputs/SelectInput"
  component={SelectInput}
  args={args}
  argTypes={argTypes}
/>

# CustomSelect

A base component extracted from [Mantine Select](https://v6.mantine.dev/core/select/) for cases where a custom dropdown is required.

## Docs

- [Figma File](https://www.figma.com/file/21uCY0czmCfb6QBjRce0I8/Input-%2F-Select?type=design&node-id=1-96&mode=design&t=Mk8RP6HsneuWzHtr-0)
- [Mantine Select Docs](https://v6.mantine.dev/core/select/)

## Examples

export const DefaultTemplate = args => <SelectInput {...args} />;

export const VariantTemplate = args => (
  <Stack>
    <SelectInput {...args} />
    <SelectInput {...args} variant="unstyled" />
  </Stack>
);

export const Default = DefaultTemplate.bind({});
