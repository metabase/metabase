# Paper

Our themed wrapper around [Mantine Paper](https://mantine.dev/core/paper/).

## Docs

- [Mantine Paper Docs](https://mantine.dev/core/paper/)

## Examples

export const Template = args => (
  <Paper {...args}>
    <Text>
      Having small touches of colour makes it more colourful than having the
      whole thing in colour
    </Text>
  </Paper>
);

export const Default = args => <Template {...args} />;
