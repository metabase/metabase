import { Canvas, Story, Meta } from "@storybook/addon-docs";
import {NestedItemPicker} from "./NestedItemPicker";

<Meta
  title="Common/NestedItemPicker"
  component={NestedItemPicker}
  args={{}}
  argTypes={{}}
/>

export const initialState = new Array(3).fill(0).map((_, index, arr) => ({
  items: new Array(10).fill(0).map((__, i) => ({
    id: `${index}-${i}`,
    name: `level-${index}-item-${i}`,
    model: i < 5 ? "collection" : "dashboard"
  })),
  selectedItem: index === arr.length - 1 ? {} : {
    id: `${index}-${Math.floor(Math.random() * 5)}`,
    model: "collection"
  }
}))

export const DefaultTemplate = args => (
  <NestedItemPicker
    folderModel={['collection']}
    initialState={initialState} />);

export const Default = DefaultTemplate.bind({});

<Canvas>
  <Story name="Default">{Default}</Story>
</Canvas>
