  align: "left",
  order: 1,
  underline: false,
  truncate: false,
};

  align: {
    options: ["left", "center", "right"],
    control: { type: "inline-radio" },
  },
  order: {

    control: { type: "inline-radio" },
  },
  underline: {
    control: { type: "boolean" },
  },
  truncate: {
    control: { type: "boolean" },
  },
};

  title="Typography/Title"

  args={args}

## Examples

export const DefaultTemplate = args => <Title {...args}>Header</Title>;

export const SizeTemplate = args => (
  <Stack>
    <Title {...args} order={1}>
      Header 1
    </Title>
    <Title {...args} order={2}>
      Header 2
    </Title>
    <Title {...args} order={3}>
      Header 3
    </Title>
    <Title {...args} order={4}>
      Header 4
    </Title>
  </Stack>
);

export const TruncatedTemplate = args => (
  <Stack>
    <Box w="6rem">
      <Title {...args} order={1}>
        Header 1
      </Title>
    </Box>
    <Box w="5rem">
      <Title {...args} order={2}>
        Header 2
      </Title>
    </Box>
    <Box w="4rem">
      <Title {...args} order={3}>
        Header 3
      </Title>
    </Box>
    <Box w="3.5rem">
      <Title {...args} order={4}>
        Header 4
      </Title>
    </Box>
  </Stack>
);

export const Default = DefaultTemplate.bind({});

<Canvas>
  <Story name="Default">{Default}</Story>
</Canvas>

export const Sizes = SizeTemplate.bind({});
